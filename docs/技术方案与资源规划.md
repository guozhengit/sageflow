## 4. 部署方案

### 4.1 Docker Compose 配置

```yaml
# docker-compose.yml
version: '3.8'

services:
  # LLM 服务 (Ollama)
  ollama:
    image: ollama/ollama:latest
    container_name: sageflow-ollama
    ports:
      - "11434:11434"
    volumes:
      - ollama_data:/root/.ollama
    deploy:
      resources:
        limits:
          memory: 4G
          cpus: '2.0'
    restart: unless-stopped

  # 向量数据库 (Qdrant)
  qdrant:
    image: qdrant/qdrant:v1.8.0
    container_name: sageflow-qdrant
    ports:
      - "6333:6333"
      - "6334:6334"
    volumes:
      - qdrant_data:/qdrant/storage
    deploy:
      resources:
        limits:
          memory: 2G
          cpus: '0.5'
    restart: unless-stopped

  # PostgreSQL
  postgres:
    image: postgres:16-alpine
    container_name: sageflow-postgres
    ports:
      - "5432:5432"
    environment:
      POSTGRES_DB: sageflow
      POSTGRES_USER: sageflow
      POSTGRES_PASSWORD: ${DB_PASSWORD:-sageflow123}
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./postgres/init.sql:/docker-entrypoint-initdb.d/init.sql
    deploy:
      resources:
        limits:
          memory: 1G
          cpus: '0.5'
    restart: unless-stopped

  # Redis
  redis:
    image: redis:7-alpine
    container_name: sageflow-redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    deploy:
      resources:
        limits:
          memory: 512M
          cpus: '0.25'
    restart: unless-stopped

  # Backend (FastAPI)
  backend:
    build: ./backend
    container_name: sageflow-backend
    ports:
      - "8000:8000"
    environment:
      - DATABASE_URL=postgresql://sageflow:${DB_PASSWORD:-sageflow123}@postgres:5432/sageflow
      - QDRANT_URL=http://qdrant:6333
      - REDIS_URL=redis://redis:6379/0
      - OLLAMA_URL=http://ollama:11434
      - LLM_MODEL=qwen2.5:3b
    volumes:
      - ./backend:/app
      - upload_data:/app/uploads
    depends_on:
      - postgres
      - qdrant
      - redis
      - ollama
    deploy:
      resources:
        limits:
          memory: 2G
          cpus: '1.0'
    restart: unless-stopped
    command: uvicorn app.main:app --host 0.0.0.0 --port 8000 --reload

  # Celery Worker
  celery:
    build: ./backend
    container_name: sageflow-celery
    environment:
      - DATABASE_URL=postgresql://sageflow:${DB_PASSWORD:-sageflow123}@postgres:5432/sageflow
      - QDRANT_URL=http://qdrant:6333
      - REDIS_URL=redis://redis:6379/0
      - OLLAMA_URL=http://ollama:11434
    volumes:
      - ./backend:/app
      - upload_data:/app/uploads
    depends_on:
      - postgres
      - qdrant
      - redis
      - ollama
    deploy:
      resources:
        limits:
          memory: 1G
          cpus: '0.5'
    restart: unless-stopped
    command: celery -A app.workers.celery_app worker --loglevel=info

  # Frontend (Next.js)
  frontend:
    build: ./frontend
    container_name: sageflow-frontend
    ports:
      - "3000:3000"
    environment:
      - NEXT_PUBLIC_API_URL=http://localhost:8000
    depends_on:
      - backend
    deploy:
      resources:
        limits:
          memory: 1G
          cpus: '0.5'
    restart: unless-stopped

  # Nginx
  nginx:
    image: nginx:alpine
    container_name: sageflow-nginx
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
    depends_on:
      - frontend
      - backend
    deploy:
      resources:
        limits:
          memory: 256M
          cpus: '0.25'
    restart: unless-stopped

volumes:
  ollama_data:
  qdrant_data:
  postgres_data:
  redis_data:
  upload_data:
```

### 4.2 一键启动脚本

```bash
#!/bin/bash
# start.sh - SageFlow 一键启动脚本

echo "🌿 启动 SageFlow..."

# 1. 检查 Docker
if ! command -v docker &> /dev/null; then
    echo "❌ Docker 未安装，请先安装 Docker"
    exit 1
fi

# 2. 加载模型
echo "📥 加载 Ollama 模型 (qwen2.5:3b)..."
docker exec sageflow-ollama ollama pull qwen2.5:3b || true

# 3. 启动服务
echo " 启动所有服务..."
docker-compose up -d

# 4. 等待服务就绪
echo "⏳ 等待服务就绪..."
sleep 10

# 5. 检查服务状态
echo "📊 服务状态:"
docker-compose ps

echo ""
echo "✅ SageFlow 启动完成!"
echo "📱 前端地址: http://localhost:3000"
echo "🔌 API 地址: http://localhost:8000/docs"
echo "📚 Qdrant: http://localhost:6333/dashboard"
```

# LangChain 配置优化
from langchain.vectorstores import Qdrant

## 7. 风险与应对

| 风险 | 影响 | 应对措施 |
|------|------|----------|
| 内存不足 | OOM 崩溃 | 限制容器内存，启用 swap |
| GPU 显存不足 | 模型加载失败 | 使用更小模型或 CPU 推理 |
| CPU 瓶颈 | 响应慢 | 异步处理，增加队列 |
| 磁盘空间 | 存储不足 | 定期清理，限制上传大小 |

## 8. 扩展方案

当需要升级时：

| 升级项 | 方案 |
|--------|------|
| 更大模型 | 升级到 8GB+ 显存 GPU |
| 更高并发 | 增加 CPU 核心和内存 |
| 更大存储 | 挂载外部存储或 OSS |
| 生产部署 | Kubernetes + 多副本 |
